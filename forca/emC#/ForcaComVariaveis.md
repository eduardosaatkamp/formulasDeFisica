using System;

class Program
{
    static void Main(string[] args)
    {
        // Fórmula da força: F = m * a

        // Variáveis
        double massa = 10; // Massa em kg
        double aceleracao = 5; // Aceleração em m/s^2

        // Cálculo da força
        double forca = massa * aceleracao;

        // Exibindo o resultado
        Console.WriteLine("Força = " + forca + "N");
    }
}
